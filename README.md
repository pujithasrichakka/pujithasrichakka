
#include <stdio.h>

int main()
{
    char str[100];  
    int i,count=0;
    printf("ENTER THE STRING: ");
    gets(str);
    for(i=0;str[i];i++)  
    {
    	if(str[i]==32)
    	 count++;
 	}
 	if(i>0)
 	 count++;

    return 0;
}
